487inline void CSCI441_INTERNAL::SimpleShader2::setupSimpleShader() {
488 std::string vertex_shader_src = "#version 410 core\n \
489 \n \
490 uniform mat4 model;\n \
491 uniform mat4 view;\n \
492 uniform mat4 projection;\n \
493 \n \
494 layout(location=0) in vec2 vPos;\n \
495 layout(location=1) in vec3 vColor;\n \
496 \n \
497 layout(location=0) ";
498 vertex_shader_src += (smoothShading ? "" : "flat ");
499 vertex_shader_src += "out vec4 fragColor;\n \
500 \n \
501 void main() {\n \
502 gl_Position = projection * view * model * vec4(vPos, 0.0, 1.0);\n \
503 fragColor = vec4(vColor, 1.0);\n \
504 }";
505 const char* vertexShaders[1] = { vertex_shader_src.c_str() };
506
507 std::string fragment_shader_src = "#version 410 core\n \
508 \n \
509 layout(location=0) ";
510 fragment_shader_src += (smoothShading ? "" : "flat ");
511 fragment_shader_src += " in vec4 fragColor;\n \
512 \n \
513 layout(location=0) out vec4 fragColorOut;\n \
514 \n \
515 void main() {\n \
516 fragColorOut = fragColor;\n \
517 }";
518 const char* fragmentShaders[1] = { fragment_shader_src.c_str() };
519
520 printf( "[INFO]: /--------------------------------------------------------\\\n" );
521
522 GLuint vertexShaderHandle = glCreateShader( GL_VERTEX_SHADER );
523 glShaderSource(vertexShaderHandle, 1, vertexShaders, nullptr);
524 glCompileShader(vertexShaderHandle);
525 ShaderUtils::printShaderLog(vertexShaderHandle);
526
527 GLuint fragmentShaderHandle = glCreateShader( GL_FRAGMENT_SHADER );
528 glShaderSource(fragmentShaderHandle, 1, fragmentShaders, nullptr);
529 glCompileShader(fragmentShaderHandle);
530 ShaderUtils::printShaderLog(fragmentShaderHandle);
531
532 shaderProgramHandle = glCreateProgram();
533 glAttachShader(shaderProgramHandle, vertexShaderHandle);
534 glAttachShader(shaderProgramHandle, fragmentShaderHandle);
535 glLinkProgram(shaderProgramHandle);
536 ShaderUtils::printProgramLog(shaderProgramHandle);
537
538 glDetachShader(shaderProgramHandle, vertexShaderHandle);
539 glDeleteShader(vertexShaderHandle);
540
541 glDetachShader(shaderProgramHandle, fragmentShaderHandle);
542 glDeleteShader(fragmentShaderHandle);
543
544 ShaderUtils::printShaderProgramInfo(shaderProgramHandle, true, false, false, false, true, false, true);
545
546 modelLocation = glGetUniformLocation(shaderProgramHandle, "model");
547 viewLocation = glGetUniformLocation(shaderProgramHandle, "view");
548 projectionLocation = glGetUniformLocation(shaderProgramHandle, "projection");
549
550 vertexLocation = glGetAttribLocation(shaderProgramHandle, "vPos");
551 colorLocation = glGetAttribLocation(shaderProgramHandle, "vColor");
552
553
554 glm::mat4 identity(1.0f);
555 glProgramUniformMatrix4fv(shaderProgramHandle, modelLocation, 1, GL_FALSE, &identity[0][0]);
556 glProgramUniformMatrix4fv(shaderProgramHandle, viewLocation, 1, GL_FALSE, &identity[0][0]);
557 glProgramUniformMatrix4fv(shaderProgramHandle, projectionLocation, 1, GL_FALSE, &identity[0][0]);
558
559 transformationStack.emplace_back(identity);
560
561 glUseProgram(shaderProgramHandle);
562}

564inline GLuint CSCI441_INTERNAL::SimpleShader2::registerVertexArray(const GLuint NUM_POINTS, const glm::vec2 VERTEX_POINTS[], const glm::vec3 VERTEX_COLORS[]) {
565 GLuint vaod;
566 glGenVertexArrays(1, &vaod);
567 glBindVertexArray(vaod);
568
569 GLuint vbod;
570 glGenBuffers(1, &vbod);
571 glBindBuffer(GL_ARRAY_BUFFER, vbod);
572 glBufferData(GL_ARRAY_BUFFER, sizeof(GLfloat)*NUM_POINTS*2 + sizeof(GLfloat)*NUM_POINTS*3, nullptr, GL_STATIC_DRAW);
573 glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(GLfloat)*NUM_POINTS*2, VERTEX_POINTS);
574 glBufferSubData(GL_ARRAY_BUFFER, sizeof(GLfloat)*NUM_POINTS*2, sizeof(GLfloat)*NUM_POINTS*3, VERTEX_COLORS);
575
576 glEnableVertexAttribArray(vertexLocation);
577 glVertexAttribPointer(vertexLocation, 2, GL_FLOAT, GL_FALSE, 0, (void*)0);
578
579 glEnableVertexAttribArray(colorLocation);
580 glVertexAttribPointer(colorLocation, 3, GL_FLOAT, GL_FALSE, 0, (void*)(sizeof(GLfloat)*NUM_POINTS*2));
581
582 return vaod;
583}

596inline void CSCI441_INTERNAL::SimpleShader2::pushTransformation(const glm::mat4& TRANSFORMATION_MATRIX) {
597 transformationStack.emplace_back(TRANSFORMATION_MATRIX);
598
599 modelMatrix *= TRANSFORMATION_MATRIX;

603inline void CSCI441_INTERNAL::SimpleShader2::popTransformation() {
604 // ensure there is a transformation stack to pop off
605 // never let the original identity matrix pop off
606 if( transformationStack.size() > 1 ) {
607 transformationStack.pop_back();
608
609 modelMatrix = glm::mat4(1.0f);
610 for( auto tMtx : transformationStack ) {
611 modelMatrix *= tMtx;
612 }
613 glProgramUniformMatrix4fv( shaderProgramHandle, modelLocation, 1, GL_FALSE, &modelMatrix[0][0] );
614 }
615}

617inline void CSCI441_INTERNAL::SimpleShader2::resetTransformationMatrix() {
618 modelMatrix = glm::mat4(1.0f);
619 transformationStack.clear();
620 transformationStack.emplace_back(modelMatrix);
621 glProgramUniformMatrix4fv(shaderProgramHandle, modelLocation, 1, GL_FALSE, &modelMatrix[0][0]);
622}

646 uniform vec3 lightColor;\n \
647 uniform vec3 lightPosition;\n \
648 uniform vec3 materialColor;\n \
649 \n \
650 layout(location=0) in vec3 vPos;\n \
651 layout(location=2) in vec3 vNormal;\n \
652 \n \
653 layout(location=0) ";
654 vertex_shader_src += (smoothShading ? "" : "flat ");
655 vertex_shader_src += "out vec4 fragColor;\n \
656 \n \
657 void main() {\n \
658 gl_Position = projection * view * model * vec4(vPos, 1.0);\n \
659 \n \
660 vec3 vertexEye = (view * model * vec4(vPos, 1.0)).xyz;\n \
661 vec3 lightEye = (view * vec4(lightPosition, 1.0)).xyz;\n \
662 vec3 lightVec = normalize( lightEye - vertexEye );\n \
663 vec3 normalVec = normalize( normalMtx * vNormal );\n \
664 float sDotN = max(dot(lightVec, normalVec), 0.0);\n \
665 vec3 diffColor = lightColor * materialColor * sDotN;\n \
666 vec3 ambColor = materialColor * 0.3;\

679 fragment_shader_src += " in vec4 fragColor;\n \
680 \n \
681 layout(location=0) out vec4 fragColorOut;\n \
682 \n \
683 void main() {\n \
684 if(useLighting == 1) {\n \
685 fragColorOut = fragColor;\n \
686 } else {\n \
687 fragColorOut = vec4(materialColor, 1.0f);\n \
688 }\n \
689 }";

719 viewLocation = glGetUniformLocation(shaderProgramHandle, "view");

810inline void CSCI441_INTERNAL::SimpleShader3::popTransformation() {
811 // ensure there is a transformation stack to pop off
812 // never let the original identity matrix pop off
813 if( transformationStack.size() > 1 ) {
814 transformationStack.pop_back();
815
816 modelMatrix = glm::mat4(1.0f);
817 for( auto tMtx : transformationStack ) {
818 modelMatrix *= tMtx;
819 }
820 glProgramUniformMatrix4fv( shaderProgramHandle, modelLocation, 1, GL_FALSE, &modelMatrix[0][0] );
821
822 setNormalMatrix();
823 }
