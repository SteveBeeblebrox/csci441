98inline CSCI441::UniformBufferObject::UniformBufferObject(const char* UNIFORM_BLOCK_NAME, std::initializer_list<const char*> uniformNamesList) {
99 _blockName = (char*)UNIFORM_BLOCK_NAME;
100
101 _numUniforms = uniformNamesList.size();
102
103 for(const auto &uniformName : uniformNamesList ) {
104 char* un = (char*)malloc(strlen(uniformName) * sizeof(char));
105 strcpy(un, uniformName);
106 _uniformNames.push_back(un);
107 }
108
109 _uniformIndices = (GLuint*)malloc(_numUniforms * sizeof(GLuint));
110 _uniformOffsets = (GLint*)malloc(_numUniforms * sizeof(GLint));
111
112 _blockSize = 0;
113 _buffer = nullptr;
114 _bindingPoint = 0;
115 _ubod = 0;
116}

129[[maybe_unused]]
130inline void CSCI441::UniformBufferObject::setupWithShaderProgram( ShaderProgram *shaderProgram, GLuint bindingPoint ) {
131 _blockSize = shaderProgram->getUniformBlockSize( _blockName );
132 _buffer = (GLubyte*)malloc( _blockSize );
133
134 glGetUniformIndices(shaderProgram->getShaderProgramHandle(), _numUniforms, &_uniformNames[0], _uniformIndices);
135 glGetActiveUniformsiv(shaderProgram->getShaderProgramHandle(), _numUniforms, _uniformIndices, GL_UNIFORM_OFFSET, _uniformOffsets);
136
137 glGenBuffers(1, &_ubod);
138 bindBuffer();
139 glBufferData(GL_UNIFORM_BUFFER, _blockSize, nullptr, GL_DYNAMIC_DRAW);
140
141 _bindingPoint = bindingPoint;
142 glBindBufferBase(GL_UNIFORM_BUFFER, _bindingPoint, _ubod);
143 shaderProgram->setUniformBlockBinding(_blockName, _bindingPoint);
144}

147inline void CSCI441::UniformBufferObject::copyToOffset( unsigned int offset, void* addr, size_t len ) {
148 if(offset < _numUniforms) {
149 memcpy(_buffer + _uniformOffsets[offset], addr, len);

151 fprintf(stderr, "[ERROR]: Offset %d exceeds size of Uniform Block %s which is %d\n", offset, _blockName, _numUniforms);

155[[maybe_unused]]
156inline void CSCI441::UniformBufferObject::copyToBuffer( const char* UNIFORM_NAME, void* addr, size_t len ) {
157 bool found = false;
158 for(GLuint i = 0; i < _numUniforms; i++) {
159 if( strcmp(_uniformNames[i], UNIFORM_NAME) == 0 ) {
160 memcpy( _buffer + _uniformOffsets[i], addr, len );
161 found = true;
162 break;
163 }
164 }
165 if(!found) {
166 fprintf(stderr, "[ERROR]: Uniform Name \"%s\" not found within Uniform Block \"%s\"\n", UNIFORM_NAME, _blockName);
167 }
168}
