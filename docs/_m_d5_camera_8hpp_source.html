131inline bool CSCI441::MD5Camera::_loadMD5CameraFromFile(
132 const char * const MD5CAMERA_FILE,
133 const GLboolean INFO,
134 const GLboolean ERRORS
135) {
136 if ( INFO ) fprintf( stdout, "[.md5camera]: -=-=-=-=-=-=-=- BEGIN %s Info -=-=-=-=-=-=-=- \n", MD5CAMERA_FILE );
137
138 time_t start, end;
139 time(&start);
140
141 std::ifstream md5CameraFile(MD5CAMERA_FILE);
142 if( !md5CameraFile.is_open() ) {
143 if (ERRORS) fprintf( stderr, "[.md5camera]: [ERROR]: Could not open \"%s\"\n", MD5CAMERA_FILE );
144 if ( INFO ) fprintf( stdout, "[.md5camera]: -=-=-=-=-=-=-=- END %s Info -=-=-=-=-=-=-=- \n", MD5CAMERA_FILE );
145 return false;
146 }
147
148 std::string sectionLabel, commandLineStr, brace;
149 int md5version;
150 Frame frame = { glm::vec3(0.0f, 0.0f, 0.0f), glm::vec3(0.0f, 0.0f, 0.0f), 45.0f };
151
152 // MD5Version 10
153 md5CameraFile >> sectionLabel >> md5version;
154 if(sectionLabel != "MD5Version" || md5version != 10) {
155 if (ERRORS) fprintf (stderr, "[.md5camera]: [ERROR]: improper MD5Camera version found \"%s %d\"\n", sectionLabel.c_str(), md5version );
156 if ( INFO ) fprintf( stdout, "[.md5camera]: -=-=-=-=-=-=-=- END %s Info -=-=-=-=-=-=-=- \n", MD5CAMERA_FILE );
157 return false;
158 }
159
160 // commandline "string"
161 md5CameraFile >> sectionLabel;
162 getline(md5CameraFile, commandLineStr);
163
164 for(unsigned short i = 0; i < 5; i++) {
165 md5CameraFile >> sectionLabel;
166 if(sectionLabel == "numFrames") {
167 // numFrames <integer>
168 md5CameraFile >> _numFrames;
169 _frames = (Frame*)malloc(sizeof(Frame) * _numFrames);
170 } else if(sectionLabel == "frameRate") {
171 // frameRate <integer>
172 md5CameraFile >> _frameRate;
173 } else if(sectionLabel == "numCuts") {
174 // numCuts <integer>
175 md5CameraFile >> _numCuts;
176 _cutPositions = (GLuint*)malloc(sizeof(GLuint) * _numCuts);
177 } else if(sectionLabel == "cuts") {
178 // cuts {
179 // [frameNumber]
180 // [frameNumber]
181 // }
182 md5CameraFile >> brace;
183 for(unsigned int cutNumber = 0; cutNumber < _numCuts; cutNumber++) {
184 md5CameraFile >> _cutPositions[cutNumber];
185 }
186 md5CameraFile >> brace;
187 _currentFrameIndex = _cutPositions[_currentCutIndex];
188 } else if(sectionLabel == "camera") {
189 // camera {
190 // ( [x] [y] [z] ) ( [orientation] ) [FOV]
191 // }
192 md5CameraFile >> brace;
193 for(unsigned int frameNumber = 0; frameNumber < _numFrames; frameNumber++) {
194 md5CameraFile >> brace >> frame.cameraPosition.x >> frame.cameraPosition.y >> frame.cameraPosition.z >> brace;
195 md5CameraFile >> brace >> frame.cameraQuaternion.x >> frame.cameraQuaternion.y >> frame.cameraQuaternion.z >> brace;
196 md5CameraFile >> frame.fieldOfView;
197 }
198 md5CameraFile >> brace;
199 } else {
200 if (ERRORS) fprintf( stderr, "[.md5camera]: [ERROR]: unknown section label found \"%s\"\n", sectionLabel.c_str() );
201 if ( INFO ) fprintf( stdout, "[.md5camera]: -=-=-=-=-=-=-=- END %s Info -=-=-=-=-=-=-=- \n", MD5CAMERA_FILE );
202 return false;
203 }
204 }
205
206 if (INFO) {
207 printf( "[.md5camera]: Camera Stats:\n" );
208 printf( "[.md5camera]: Num Frames:\t%u\tFrame Rate:\t%u\tNum Cuts: \t%u\n", _numFrames, _frameRate, _numCuts );
209 }
210
211 time(&end);
212 double seconds = difftime( end, start );
213
214 if (INFO) {
215 printf( "[.md5camera]: Completed in %.3fs\n", seconds );
216 printf( "[.md5camera]: -=-=-=-=-=-=-=- END %s Info -=-=-=-=-=-=-=- \n\n", MD5CAMERA_FILE );
217 }
218
219 return true;
220}

222inline void CSCI441::MD5Camera::moveForward(const GLfloat unused) {
223 // prevent memory errors by checking if file loaded correctly
224 if( !_isInitialized ) return;
225
226 // clamp to current cut ? loop ? advance ?
227
228 // check if current frame is at end of overall list of frames
229 if( _currentFrameIndex == _numFrames - 1) {
230 switch(_advancementStrategy) {
231 case LOOP_ALL_CUTS:
232 // go back to start of all cuts
233 _currentCutIndex = 0;
234 case LOOP_SINGLE_CUT:
235 // go to start of current cut
236 _currentFrameIndex = _cutPositions[ _currentCutIndex ];
237 break;
238
239 case RUN_SINGLE_CUT:
240 case RUN_ALL_CUTS:
241 // do nothing, at end and not looping
242 return;
243 }
244 }
245 // check if in last cut, but can't be at the end of the last cut because then we'd be at the end of the overall frames
246 else if( _currentCutIndex == _numCuts - 1 ) {
247 // in the middle of the last cut, move forward
248 _currentFrameIndex++;
249 }
250 // otherwise not on final overall frame and not in the last cut
251 else {
252 // check if at end of current cut
253 if( _currentFrameIndex == _cutPositions[_currentCutIndex + 1] - 1 ) {
254 switch(_advancementStrategy) {
255 case RUN_ALL_CUTS:
256 case LOOP_ALL_CUTS:
257 // go to next cut
258 _currentCutIndex++;
259 case LOOP_SINGLE_CUT:
260 // go to start of current cut
261 _currentFrameIndex = _cutPositions[ _currentCutIndex ];
262 break;
263
264 case RUN_SINGLE_CUT:
265 // do nothing, at end and not looping nor advancing
266 return;
267 }
268 } else {
269 // in the middle of a cut, move forward
270 _currentFrameIndex++;
271 }
272 }
273 _updateCameraAttributesForCurrentFrame();
274}

276inline void CSCI441::MD5Camera::moveBackward(const GLfloat unused) {
277 // prevent memory errors by checking if file loaded correctly
278 if( !_isInitialized ) return;
279
280 // clamp to current cut ? loop ? advance ?
281
282 // check if current frame is at beginning of overall list of frames
283 if( _currentFrameIndex == 0) {
284 switch(_advancementStrategy) {
285 case LOOP_ALL_CUTS:
286 // go back to end of all frames
287 _currentCutIndex = _numFrames-1;
288 break;
289
290 case LOOP_SINGLE_CUT:
291 if( _numCuts == 1 ) {
292 // go back to end of all frames
293 _currentCutIndex = _numFrames-1;
294 } else {
295 // go to end of current cut, which is the frame before the next cut
296 _currentFrameIndex = _cutPositions[ _currentCutIndex+1 ] - 1;
297 }
298 break;
299
300 case RUN_SINGLE_CUT:
301 case RUN_ALL_CUTS:
302 // do nothing, at end and not looping
303 return;
304 }
305 }
306 // check if in first cut, but can't be at the end of the first cut because then we'd be at the end of the overall frames
307 else if( _currentCutIndex == 0 ) {
308 // in the middle of the first cut, move backward
309 _currentFrameIndex--;
310 }
311 // otherwise not on initial overall frame and not in the first cut
312 else {
313 // check if at beginning of current cut
314 if( _currentFrameIndex == _cutPositions[_currentCutIndex] ) {
315 switch(_advancementStrategy) {
316 case RUN_ALL_CUTS:
317 case LOOP_ALL_CUTS:
318 // go to previous cut
319 _currentCutIndex--;
320 case LOOP_SINGLE_CUT:
321 // go to end of current cut, which is the frame before the next cut
322 _currentFrameIndex = _cutPositions[ _currentCutIndex+1 ] - 1;
323 break;
324
325 case RUN_SINGLE_CUT:
326 // do nothing, at end and not looping nor advancing
327 return;
328 }
329 } else {
330 // in the middle of a cut, move backward
331 _currentFrameIndex--;
332 }
333 }
334 _updateCameraAttributesForCurrentFrame();
335}

337inline void CSCI441::MD5Camera::_updateCameraAttributesForCurrentFrame() {
338 // get and set camera position for current frame
339 mCameraPosition = _frames[_currentFrameIndex].cameraPosition;
340
341 // get and set camera orientation for current frame
342 // compute W of quaternion
343 glm::vec4 q(_frames[_currentFrameIndex].cameraQuaternion, 0.0f);
344 GLfloat t = 1.0f - (q.x * q.x) - (q.y * q.y) - (q.z * q.z);
345 if (t < 0.0f)
346 q.w = 0.0f;
347 else
348 q.w = -glm::sqrt(t);
349
350 // set direction and look at point
351 glm::vec3 defaultCameraDirection(0.0f, 0.0f, -1.0f);
352 glm::vec4 inverseQ(-q.x, -q.y, -q.z, q.w);
353 glm::normalize(inverseQ);
354
355 glm::vec4 tmp( (q.w * defaultCameraDirection.x) + (q.y * defaultCameraDirection.z) - (q.z * defaultCameraDirection.y),
356 (q.w * defaultCameraDirection.y) + (q.z * defaultCameraDirection.x) - (q.x * defaultCameraDirection.z),
357 (q.w * defaultCameraDirection.z) + (q.x * defaultCameraDirection.y) - (q.y * defaultCameraDirection.x),
358 -(q.w * defaultCameraDirection.x) - (q.y * defaultCameraDirection.y) - (q.z * defaultCameraDirection.z) );
359 glm::vec4 rotatedCameraDirection((tmp.x * inverseQ.w) + (tmp.w * inverseQ.x) + (tmp.y * inverseQ.z) - (tmp.z * inverseQ.y),
360 (tmp.y * inverseQ.w) + (tmp.w * inverseQ.y) + (tmp.z * inverseQ.x) - (tmp.x * inverseQ.z),
361 (tmp.z * inverseQ.w) + (tmp.w * inverseQ.z) + (tmp.x * inverseQ.y) - (tmp.y * inverseQ.x),
362 (tmp.w * inverseQ.w) - (tmp.x * inverseQ.x) - (tmp.y * inverseQ.y) - (tmp.z * inverseQ.z) );
363 mCameraDirection = glm::vec3( rotatedCameraDirection.x, rotatedCameraDirection.y, rotatedCameraDirection.z );
364 mCameraLookAtPoint = mCameraPosition + mCameraDirection;
365
366 // set up vector
367 glm::vec3 defaultCameraUpVector(0.0f, 1.0f, 0.0f);
368 glm::vec4 tmp2( (q.w * defaultCameraUpVector.x) + (q.y * defaultCameraUpVector.z) - (q.z * defaultCameraUpVector.y),
369 (q.w * defaultCameraUpVector.y) + (q.z * defaultCameraUpVector.x) - (q.x * defaultCameraUpVector.z),
370 (q.w * defaultCameraUpVector.z) + (q.x * defaultCameraUpVector.y) - (q.y * defaultCameraUpVector.x),
371 -(q.w * defaultCameraUpVector.x) - (q.y * defaultCameraUpVector.y) - (q.z * defaultCameraUpVector.z) );
372 glm::vec4 rotatedCameraUpVector((tmp.x * inverseQ.w) + (tmp.w * inverseQ.x) + (tmp.y * inverseQ.z) - (tmp.z * inverseQ.y),
373 (tmp.y * inverseQ.w) + (tmp.w * inverseQ.y) + (tmp.z * inverseQ.x) - (tmp.x * inverseQ.z),
374 (tmp.z * inverseQ.w) + (tmp.w * inverseQ.z) + (tmp.x * inverseQ.y) - (tmp.y * inverseQ.x),
375 (tmp.w * inverseQ.w) - (tmp.x * inverseQ.x) - (tmp.y * inverseQ.y) - (tmp.z * inverseQ.z) );
376 mCameraUpVector = glm::vec3(rotatedCameraUpVector.x, rotatedCameraUpVector.y, rotatedCameraUpVector.z);
377
378 // compute and set view matrix
379 computeViewMatrix();
380
381 // get and set field of view for perspective projection matrix
382 _fovy = _frames[_currentFrameIndex].fieldOfView;
383 mProjectionMatrix = glm::perspective(_fovy, _aspectRatio, _nearClipPlane, _farClipPlane);
384}
