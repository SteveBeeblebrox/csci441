578 GLint major, minor;
579 glGetIntegerv(GL_MAJOR_VERSION, &major);
580 glGetIntegerv(GL_MINOR_VERSION, &minor);
581
582 if( sDEBUG ) printf( "\n[INFO]: /--------------------------------------------------------\\\n");
583
584 /* compile each one of our shaders */
585 if( strcmp( vertexShaderFilename, "" ) != 0 ) {
586 if( sDEBUG ) printf( "[INFO]: | Vertex Shader: %39s |\n", vertexShaderFilename );
587 mVertexShaderHandle = CSCI441_INTERNAL::ShaderUtils::compileShader(vertexShaderFilename, GL_VERTEX_SHADER );
588 } else {
589 mVertexShaderHandle = 0;
590 }
591
592 if( strcmp( tesselationControlShaderFilename, "" ) != 0 ) {
593 if( sDEBUG ) printf( "[INFO]: | Tess Control Shader: %33s |\n", tesselationControlShaderFilename );
594 if( major < 4 ) {
595 printf( "[ERROR]:| TESSELATION SHADER NOT SUPPORTED!! UPGRADE TO v4.0+ |\n" );
596 mTesselationControlShaderHandle = 0;
597 } else {
598 mTesselationControlShaderHandle = CSCI441_INTERNAL::ShaderUtils::compileShader(tesselationControlShaderFilename, GL_TESS_CONTROL_SHADER );
599 }
600 } else {
601 mTesselationControlShaderHandle = 0;
602 }
603
604 if( strcmp( tesselationEvaluationShaderFilename, "" ) != 0 ) {
605 if( sDEBUG ) printf( "[INFO]: | Tess Evaluation Shader: %30s |\n", tesselationEvaluationShaderFilename );
606 if( major < 4 ) {
607 printf( "[ERROR]:| TESSELATION SHADER NOT SUPPORTED!! UPGRADE TO v4.0+ |\n" );
608 mTesselationEvaluationShaderHandle = 0;
609 } else {
610 mTesselationEvaluationShaderHandle = CSCI441_INTERNAL::ShaderUtils::compileShader(tesselationEvaluationShaderFilename, GL_TESS_EVALUATION_SHADER );
611 }
612 } else {
613 mTesselationEvaluationShaderHandle = 0;
614 }
615
616 if( strcmp( geometryShaderFilename, "" ) != 0 ) {
617 if( sDEBUG ) printf( "[INFO]: | Geometry Shader: %37s |\n", geometryShaderFilename );
618 if( major < 3 || (major == 3 && minor < 2) ) {
619 printf( "[ERROR]:| GEOMETRY SHADER NOT SUPPORTED!!! UPGRADE TO v3.2+ |\n" );
620 mGeometryShaderHandle = 0;
621 } else {
622 mGeometryShaderHandle = CSCI441_INTERNAL::ShaderUtils::compileShader(geometryShaderFilename, GL_GEOMETRY_SHADER );
623 }
624 } else {
625 mGeometryShaderHandle = 0;
626 }
627
628 if( strcmp( fragmentShaderFilename, "" ) != 0 ) {
629 if( sDEBUG ) printf( "[INFO]: | Fragment Shader: %37s |\n", fragmentShaderFilename );
630 mFragmentShaderHandle = CSCI441_INTERNAL::ShaderUtils::compileShader(fragmentShaderFilename, GL_FRAGMENT_SHADER );
631 } else {
632 mFragmentShaderHandle = 0;
633 }
634 /* get a handle to a shader program */
635 mShaderProgramHandle = glCreateProgram();
636
637 /* if program is separable, make it so */
638 if( isSeparable ) {
639 glProgramParameteri(mShaderProgramHandle, GL_PROGRAM_SEPARABLE, GL_TRUE );
640 }
641

643 if(mVertexShaderHandle != 0 ) {
644 glAttachShader(mShaderProgramHandle, mVertexShaderHandle );

657 }

663
664 /* check the program log */
665 CSCI441_INTERNAL::ShaderUtils::printProgramLog(mShaderProgramHandle );
666
667 /* detach & delete the vertex and fragment shaders to the shader program */
668 if(mVertexShaderHandle != 0 ) {
669 glDetachShader(mShaderProgramHandle, mVertexShaderHandle );
670 glDeleteShader(mVertexShaderHandle );
671 }
672 if(mTesselationControlShaderHandle != 0 ) {
673 glDetachShader(mShaderProgramHandle, mTesselationControlShaderHandle );
674 glDeleteShader(mTesselationControlShaderHandle );
675 }
676 if(mTesselationEvaluationShaderHandle != 0 ) {
677 glDetachShader(mShaderProgramHandle, mTesselationEvaluationShaderHandle );
678 glDeleteShader(mTesselationEvaluationShaderHandle );

682 glDeleteShader(mGeometryShaderHandle );

686 glDeleteShader(mFragmentShaderHandle );

690 mpUniformLocationsMap = new std::map<std::string, GLint>();

695 char name[64];
696 int max_length = 64;

698 int size = 0;
699 GLenum type;

701 GLint location = -1;

703 for(int j = 0; j < size; j++) {
704 char long_name[64];
705 sprintf(long_name, "%s[%i]", name, j);
706 location = glGetUniformLocation(mShaderProgramHandle, long_name);
707 }
708 } else {
709 location = glGetUniformLocation(mShaderProgramHandle, name);
710 }
711 mpUniformLocationsMap->emplace(name, location );
712 }
713 }
714
715 // map attributes
716 mpAttributeLocationsMap = new std::map<std::string, GLint>();
717 GLint numAttributes;
718 glGetProgramiv(mShaderProgramHandle, GL_ACTIVE_ATTRIBUTES, &numAttributes );
719 if( numAttributes > 0 ) {
720 for(GLuint i = 0; i < numAttributes; i++) {
721 char name[64];
722 int max_length = 64;
723 int actual_length = 0;
724 int size = 0;
725 GLenum type;
726 glGetActiveAttrib(mShaderProgramHandle, i, max_length, &actual_length, &size, &type, name );
727 GLint location = -1;
728 if( size > 1 ) {
729 for( int j = 0; j < size; j++ ) {
730 char long_name[64];
731 sprintf( long_name, "%s[%i]", name, j );
732 location = glGetAttribLocation(mShaderProgramHandle, long_name );
733 }
734 } else {
735 location = glGetAttribLocation(mShaderProgramHandle, name );
736 }
737 mpAttributeLocationsMap->emplace(name, location );
738 }
739 }
740
741 GLint separable = GL_FALSE;
742 glGetProgramiv(mShaderProgramHandle, GL_PROGRAM_SEPARABLE, &separable );
743
744 if( sDEBUG ) printf( "[INFO]: | Program Separable: %35s |\n", (separable ? "Yes" : "No"));
745
746 GLint linkStatus;
747 glGetProgramiv(mShaderProgramHandle, GL_LINK_STATUS, &linkStatus );
748
749 /* print shader info for uniforms & attributes */
750 if(linkStatus == 1) {
751 CSCI441_INTERNAL::ShaderUtils::printShaderProgramInfo(mShaderProgramHandle, mVertexShaderHandle != 0,
752 mTesselationControlShaderHandle != 0,
753 mTesselationEvaluationShaderHandle != 0,
754 mGeometryShaderHandle != 0, mFragmentShaderHandle != 0,
755 false, true);
756 }
757 /* return handle */
758 return mShaderProgramHandle != 0;
759}
