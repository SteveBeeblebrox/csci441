352inline void CSCI441::OpenGLEngine::mSetupGLFW() {
353 // set what function to use when registering errors
354 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
355 // all other GLFW calls must be performed after GLFW has been initialized
356 glfwSetErrorCallback(mErrorCallback);
357
358 // initialize GLFW
359 if( !glfwInit() ) {
360 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
361 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
362 } else {
363 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
364
365 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
366 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
367 glfwWindowHint( GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
368 glfwWindowHint( GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
369 glfwWindowHint( GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
370 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
371
372 // create a window for a given size, with a given title
373 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
374 if( !mpWindow ) { // if the window could not be created, NULL is returned
375 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
376 glfwTerminate();
377 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
378 } else {
379 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
380 glfwMakeContextCurrent(mpWindow); // make the created window the current window
381 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
382
383 glfwSetWindowUserPointer(mpWindow, (void*)this);
384 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
385 }
386 }
387}

414inline void CSCI441::OpenGLEngine::shutdown() {
415 if( !_isCleanedUp ) {
416 if (DEBUG) fprintf(stdout, "\n[INFO]: Shutting down.......\n");
417 mCleanupShaders(); // delete shaders from GPU
418 mCleanupBuffers(); // delete VAOs/VBOs from GPU
419 mCleanupTextures(); // delete textures from GPU
420 mCleanupScene(); // delete scene info from CPU
421 mCleanupOpenGL(); // cleanup anything OpenGL related
422 _cleanupGLEW(); // cleanup anything GLEW related
423 mCleanupGLFW(); // shut down GLFW to clean up our context
424 if (DEBUG) fprintf(stdout, "[INFO]: ..shut down complete!\n");
425 _isCleanedUp = true;
426 _isInitialized = false;
427 }
