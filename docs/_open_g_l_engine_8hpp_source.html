357 if (DEBUG) fprintf(stdout, "\n[INFO]: Setup complete\n");
358 }

361inline void CSCI441::OpenGLEngine::mSetupGLFW() {
362 // set what function to use when registering errors
363 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
364 // all other GLFW calls must be performed after GLFW has been initialized
365 glfwSetErrorCallback(mErrorCallback);
366
367 // initialize GLFW
368 if( !glfwInit() ) {
369 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
370 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
371 } else {
372 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
373
374 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
375 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
376 glfwWindowHint( GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
377 glfwWindowHint( GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
378 glfwWindowHint( GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
379 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
380
381 // create a window for a given size, with a given title
382 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
383 if( !mpWindow ) { // if the window could not be created, NULL is returned
384 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
385 glfwTerminate();
386 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
387 } else {
388 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
389 glfwMakeContextCurrent(mpWindow); // make the created window the current window
390 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
391
392 glfwSetWindowUserPointer(mpWindow, (void*)this);
393 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
394 }
395 }
396}

423inline void CSCI441::OpenGLEngine::shutdown() {
424 if( !_isCleanedUp ) {
425 if (DEBUG) fprintf(stdout, "\n[INFO]: Shutting down.......\n");
426 mCleanupShaders(); // delete shaders from GPU
427 mCleanupBuffers(); // delete VAOs/VBOs from GPU
428 mCleanupTextures(); // delete textures from GPU
429 mCleanupScene(); // delete scene info from CPU
430 mCleanupOpenGL(); // cleanup anything OpenGL related
431 _cleanupGLEW(); // cleanup anything GLEW related
432 mCleanupGLFW(); // shut down GLFW to clean up our context
433 if (DEBUG) fprintf(stdout, "[INFO]: ..shut down complete!\n");
434 _isCleanedUp = true;
435 _isInitialized = false;
436 }
