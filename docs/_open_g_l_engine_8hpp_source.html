351inline void CSCI441::OpenGLEngine::mSetupGLFW() {
352 // set what function to use when registering errors
353 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
354 // all other GLFW calls must be performed after GLFW has been initialized
355 glfwSetErrorCallback(mErrorCallback);
356
357 // initialize GLFW
358 if( !glfwInit() ) {
359 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
360 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
361 } else {
362 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
363
364 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
365 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
366 glfwWindowHint( GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
367 glfwWindowHint( GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
368 glfwWindowHint( GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
369 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
370
371 // create a window for a given size, with a given title
372 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
373 if( !mpWindow ) { // if the window could not be created, NULL is returned
374 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
375 glfwTerminate();
376 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
377 } else {
378 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
379 glfwMakeContextCurrent(mpWindow); // make the created window the current window
380 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
381
382 glfwSetWindowUserPointer(mpWindow, (void*)this);
383 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
384 }
385 }
386}

413inline void CSCI441::OpenGLEngine::shutdown() {
414 if( !_isCleanedUp ) {
415 if (DEBUG) fprintf(stdout, "\n[INFO]: Shutting down.......\n");
416 mCleanupShaders(); // delete shaders from GPU
417 mCleanupBuffers(); // delete VAOs/VBOs from GPU
418 mCleanupTextures(); // delete textures from GPU
419 mCleanupScene(); // delete scene info from CPU
420 mCleanupOpenGL(); // cleanup anything OpenGL related
421 _cleanupGLEW(); // cleanup anything GLEW related
422 mCleanupGLFW(); // shut down GLFW to clean up our context
423 if (DEBUG) fprintf(stdout, "[INFO]: ..shut down complete!\n");
424 _isCleanedUp = true;
425 _isInitialized = false;
