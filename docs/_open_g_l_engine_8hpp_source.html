376inline void CSCI441::OpenGLEngine::initialize() {
377 if( !_isInitialized ) {
378 mSetupGLFW(); // initialize GLFW and set up a window
379 _setupGLFunctions(); // create OpenGL function pointers
380 mSetupOpenGL(); // create the OpenGL context
381
382 // get OpenGL context information
383 if( DEBUG ) {
384 // if wanting debug information with Version 4.3 or higher
385 if( mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3) ) {
386 // check if debug context was created
387 int flags; glGetIntegerv(GL_CONTEXT_FLAGS, &flags);
388 if (flags & GL_CONTEXT_FLAG_DEBUG_BIT) {
389 // register callback to synchronously print any debug messages without having to call glGetError()
390 glEnable(GL_DEBUG_OUTPUT);
391 glEnable(GL_DEBUG_OUTPUT_SYNCHRONOUS);
392 glDebugMessageCallback(mDebugMessageCallback, nullptr);
393 glDebugMessageControl( GL_DONT_CARE, GL_DONT_CARE, GL_DONT_CARE, 0, nullptr, GL_TRUE );
394 }
395 }
396
397 CSCI441::OpenGLUtils::printOpenGLInfo();
398 }
399
400 mSetupShaders(); // transfer, compile, link shaders on GPU
401 mSetupBuffers(); // register Buffers on GPU
402 mSetupTextures(); // register Textures on GPU
403 mSetupScene(); // setup any scene specific information
404
405 _isInitialized = true;
406 _isCleanedUp = false;
407 if (DEBUG) fprintf(stdout, "\n[INFO]: Setup complete\n");
408 }
409}

411inline void CSCI441::OpenGLEngine::mSetupGLFW() {
412 // set what function to use when registering errors
413 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
414 // all other GLFW calls must be performed after GLFW has been initialized
415 glfwSetErrorCallback(mErrorCallback);
416
417 // initialize GLFW
418 if( !glfwInit() ) {
419 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
420 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
421 } else {
422 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
423
424 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, mOpenGLMajorVersion ); // request OpenGL vX.
425 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, mOpenGLMinorVersion ); // request OpenGL v .X
426 glfwWindowHint( GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
427 glfwWindowHint( GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
428 glfwWindowHint( GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
429 glfwWindowHint(GLFW_RESIZABLE, mWindowResizable ); // set if our window should be able to be resized
430
431 // if wanting debug information with Version 4.3 or higher
432 if( DEBUG
433 && (mOpenGLMajorVersion > 4 || (mOpenGLMajorVersion == 4 && mOpenGLMinorVersion >= 3)) ) {
434 glfwWindowHint(GLFW_OPENGL_DEBUG_CONTEXT, true); // request a debug context
435 }
436
437 // create a window for a given size, with a given title
438 mpWindow = glfwCreateWindow(mWindowWidth, mWindowHeight, mWindowTitle, nullptr, nullptr );
439 if( !mpWindow ) { // if the window could not be created, NULL is returned
440 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
441 glfwTerminate();
442 mErrorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
443 } else {
444 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
445 glfwMakeContextCurrent(mpWindow); // make the created window the current window
446 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
447 glfwSetInputMode(mpWindow, GLFW_LOCK_KEY_MODS, GLFW_TRUE); // track state of Caps Lock and Num Lock keys
448 glfwSetWindowUserPointer(mpWindow, (void*)this);
449 glfwSetWindowSizeCallback(mpWindow, mWindowResizeCallback);
450 }
451 }

454inline void CSCI441::OpenGLEngine::_setupGLFunctions() {
455
456#ifdef CSCI441_USE_GLEW
457 glewExperimental = GL_TRUE;
458 GLenum glewResult = glewInit(); // initialize GLEW
459
460 // check for an error
461 if( glewResult != GLEW_OK ) {
462 fprintf( stderr, "[ERROR]: Error initializing GLEW\n");
463 fprintf( stderr, "[ERROR]: %s\n", glewGetErrorString(glewResult) );
464 mErrorCode = OPENGL_ENGINE_ERROR_GLEW_INIT;
465 } else {
466 if(DEBUG) {
467 fprintf(stdout, "\n[INFO]: GLEW initialized\n");
468 fprintf(stdout, "[INFO]: Using GLEW %s\n", glewGetString(GLEW_VERSION));
469 }
470 }
471#else
472 int version = gladLoadGL(glfwGetProcAddress);
473 if(version == 0) {
474 fprintf(stderr, "Failed to initialize GLAD\n" );
475 mErrorCode = OPENGL_ENGINE_ERROR_GLAD_INIT;
476 } else {
477 if(DEBUG) {
478 // Successfully loaded OpenGL
479 fprintf(stdout, "\n[INFO]: GLAD initialized\n");
480 fprintf(stdout, "[INFO]: Loaded OpenGL %d.%d\n", GLAD_VERSION_MAJOR(version), GLAD_VERSION_MINOR(version));
481 }
482 }
483#endif
484
485 if(mErrorCode == OPENGL_ENGINE_ERROR_NO_ERROR) {
486 GLint numExtensions = 0;
487 glGetIntegerv(GL_NUM_EXTENSIONS, &numExtensions);
488 for (int i = 0; i < numExtensions; i++) {
489 _extensions.insert((const char*)glGetStringi(GL_EXTENSIONS, i) );
490 }
491 }
492}

502inline void CSCI441::OpenGLEngine::shutdown() {
503 if( !_isCleanedUp ) {
504 if (DEBUG) fprintf(stdout, "\n[INFO]: Shutting down.......\n");
505 mCleanupShaders(); // delete shaders from GPU
506 mCleanupBuffers(); // delete VAOs/VBOs from GPU
507 mCleanupTextures(); // delete textures from GPU
508 mCleanupScene(); // delete scene info from CPU
509 mCleanupOpenGL(); // cleanup anything OpenGL related
510 _cleanupGLFunctions(); // cleanup anything function pointer related
511 mCleanupGLFW(); // shut down GLFW to clean up our context
512 if (DEBUG) fprintf(stdout, "[INFO]: ..shut down complete!\n");
513 _isCleanedUp = true;
514 _isInitialized = false;
515 }
