184inline void CSCI441::OpenGLEngine::_setupGLFW() {
185 // set what function to use when registering errors
186 // this is the ONLY GLFW function that can be called BEFORE GLFW is initialized
187 // all other GLFW calls must be performed after GLFW has been initialized
188 glfwSetErrorCallback( _errorCallback );
189
190 // initialize GLFW
191 if( !glfwInit() ) {
192 fprintf( stderr, "[ERROR]: Could not initialize GLFW\n" );
193 _errorCode = OPENGL_ENGINE_ERROR_GLFW_INIT;
194 } else {
195 if(DEBUG) fprintf( stdout, "[INFO]: GLFW %d.%d.%d initialized\n", GLFW_VERSION_MAJOR, GLFW_VERSION_MINOR, GLFW_VERSION_REVISION );
196
197 glfwWindowHint( GLFW_CONTEXT_VERSION_MAJOR, _majorVersion ); // request OpenGL vX.
198 glfwWindowHint( GLFW_CONTEXT_VERSION_MINOR, _minorVersion ); // request OpenGL v.X
199 glfwWindowHint( GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE ); // request forward compatible OpenGL context
200 glfwWindowHint( GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE ); // request OpenGL Core Profile context
201 glfwWindowHint( GLFW_DOUBLEBUFFER, GLFW_TRUE ); // request double buffering
202 glfwWindowHint( GLFW_RESIZABLE, _windowResizable ); // set if our window should be able to be resized
203
204 // create a window for a given size, with a given title
205 _window = glfwCreateWindow( _windowWidth, _windowHeight, _windowTitle, nullptr, nullptr );
206 if( !_window ) { // if the window could not be created, NULL is returned
207 fprintf( stderr, "[ERROR]: GLFW Window could not be created\n" );
208 glfwTerminate();
209 _errorCode = OPENGL_ENGINE_ERROR_GLFW_WINDOW;
210 } else {
211 if(DEBUG) fprintf( stdout, "[INFO]: GLFW Window created\n" );
212 glfwMakeContextCurrent(_window); // make the created window the current window
213 glfwSwapInterval(1); // update our screen after at least 1 screen refresh
214
215 glfwSetWindowUserPointer(_window, (void*)this);
216 glfwSetWindowSizeCallback(_window, _windowResizeCallback);
217 }
218 }
219}

221inline void CSCI441::OpenGLEngine::_setupGLEW() {

225 // check for an error

246inline void CSCI441::OpenGLEngine::shutdown() {
247 if( !_isCleanedUp ) {
248 if (DEBUG) fprintf(stdout, "\n[INFO]: Shutting down.......\n");
249 _cleanupShaders(); // delete shaders from GPU
250 _cleanupBuffers(); // delete VAOs/VBOs from GPU
251 _cleanupTextures(); // delete textures from GPU
252 _cleanupScene(); // delete scene info from CPU
253 _cleanupOpenGL(); // cleanup anything OpenGL related
254 _cleanupGLEW(); // cleanup anything GLEW related
255 _cleanupGLFW(); // shut down GLFW to clean up our context
256 if (DEBUG) fprintf(stdout, "[INFO]: ..shut down complete!\n");
257 _isCleanedUp = true;
258 }
