105inline bool CSCI441::TextureUtils::loadPPM( const char *filename, int &imageWidth, int &imageHeight, unsigned char* &imageData ) {
106 FILE *fp = fopen(filename, "r");
107 int temp, maxValue;
108 fscanf(fp, "P%d", &temp);
109 if(temp != 3) {
110 fprintf(stderr, "[ERROR]: CSCI441::TextureUtils::loadPPM(): PPM file is not of correct format! (Must be P3, is P%d.)\n", temp);
111 fclose(fp);
112 return false;
113 }
114
115 //got the file header right...
116 fscanf(fp, "%d", &imageWidth);
117 fscanf(fp, "%d", &imageHeight);
118 fscanf(fp, "%d", &maxValue);
119
120 //now that we know how big it is, allocate the buffer...
121 imageData = new unsigned char[imageWidth*imageHeight*3];
122 if(!imageData) {
123 fprintf(stderr, "[ERROR]: CSCI441::TextureUtils::loadPPM(): couldn't allocate image memory. Dimensions: %d x %d.\n", imageWidth, imageHeight);
124 fclose(fp);
125 return false;
126 }
127
128 //and read the data in.
129 for(int j = 0; j < imageHeight; j++) {
130 for(int i = 0; i < imageWidth; i++) {
131 int r, g, b;
132 fscanf(fp, "%d", &r);
133 fscanf(fp, "%d", &g);
134 fscanf(fp, "%d", &b);
135
136 imageData[(j*imageWidth+i)*3+0] = r;
137 imageData[(j*imageWidth+i)*3+1] = g;
138 imageData[(j*imageWidth+i)*3+2] = b;
139 }
140 }
141
142 fclose(fp);
143 return true;
144}

151inline GLuint CSCI441::TextureUtils::loadAndRegister2DTexture( const char *filename, const GLint minFilter, const GLint magFilter, const GLint wrapS, const GLint wrapT, const GLboolean flipOnY ) {
152 int imageWidth, imageHeight, imageChannels;
153 GLuint texHandle = 0;
154 stbi_set_flip_vertically_on_load(flipOnY);
155 unsigned char *data = stbi_load( filename, &imageWidth, &imageHeight, &imageChannels, 0);
156
157 if( !data ) {
158 if( strstr(filename, ".ppm") != NULL ) {
159 loadPPM(filename, imageWidth, imageHeight, data);
160 imageChannels = 3;
161 }
162 if( !data ) {
163 printf( "[ERROR]: CSCI441::TextureUtils::loadAndRegister2DTexture(): Could not load texture \"%s\"\n", filename );
164 return texHandle;
165 }
166 }
167
168 glGenTextures(1, &texHandle );
169 glBindTexture( GL_TEXTURE_2D, texHandle );
170 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, minFilter );
171 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, magFilter );
172 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, wrapS );
173 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, wrapT );
174 const GLint STORAGE_TYPE = (imageChannels == 4 ? GL_RGBA : GL_RGB);
175 glTexImage2D( GL_TEXTURE_2D, 0, STORAGE_TYPE, imageWidth, imageHeight, 0, STORAGE_TYPE, GL_UNSIGNED_BYTE, data);
176 glGenerateMipmap(GL_TEXTURE_2D);
177 printf( "[INFO]: Successfully loaded texture \"%s\" with handle %d\n", filename, texHandle );
178
179 return texHandle;
180}
