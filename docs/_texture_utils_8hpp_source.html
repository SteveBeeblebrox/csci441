91inline bool CSCI441::TextureUtils::loadPPM( const char *filename, int &imageWidth, int &imageHeight, unsigned char* &imageData ) {
92 FILE *fp = fopen(filename, "r");
93 int temp, maxValue;
94 fscanf(fp, "P%d", &temp);
95 if(temp != 3) {
96 fprintf(stderr, "[ERROR]: CSCI441::TextureUtils::loadPPM(): PPM file is not of correct format! (Must be P3, is P%d.)\n", temp);
97 fclose(fp);
98 return false;
99 }
100
101 //got the file header right...
102 fscanf(fp, "%d", &imageWidth);
103 fscanf(fp, "%d", &imageHeight);
104 fscanf(fp, "%d", &maxValue);
105
106 //now that we know how big it is, allocate the buffer...
107 imageData = new unsigned char[imageWidth*imageHeight*3];
108 if(!imageData) {
109 fprintf(stderr, "[ERROR]: CSCI441::TextureUtils::loadPPM(): couldn't allocate image memory. Dimensions: %d x %d.\n", imageWidth, imageHeight);
110 fclose(fp);
111 return false;
112 }
113
114 //and read the data in.
115 for(int j = 0; j < imageHeight; j++) {
116 for(int i = 0; i < imageWidth; i++) {
117 int r, g, b;
118 fscanf(fp, "%d", &r);
119 fscanf(fp, "%d", &g);
120 fscanf(fp, "%d", &b);
121
122 imageData[(j*imageWidth+i)*3+0] = r;
123 imageData[(j*imageWidth+i)*3+1] = g;
124 imageData[(j*imageWidth+i)*3+2] = b;
125 }
126 }
127
128 fclose(fp);
129 return true;
130}

136inline GLuint CSCI441::TextureUtils::loadAndRegister2DTexture( const char *filename, GLint minFilter, GLint magFilter, GLint wrapS, GLint wrapT ) {
137 int imageWidth, imageHeight, imageChannels;
138 GLuint texHandle = 0;
139 stbi_set_flip_vertically_on_load(true);
140 unsigned char *data = stbi_load( filename, &imageWidth, &imageHeight, &imageChannels, 0);
141
142 if( !data ) {
143 if( strstr(filename, ".ppm") != NULL ) {
144 loadPPM(filename, imageWidth, imageHeight, data);
145 imageChannels = 3;
146 }
147 if( !data ) {
148 printf( "[ERROR]: CSCI441::TextureUtils::loadAndRegister2DTexture(): Could not load texture \"%s\"\n", filename );
149 return texHandle;
150 }
151 }
152
153 glGenTextures(1, &texHandle );
154 glBindTexture( GL_TEXTURE_2D, texHandle );
155 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, minFilter );
156 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, magFilter );
157 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, wrapS );
158 glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, wrapT );
159 const GLint STORAGE_TYPE = (imageChannels == 4 ? GL_RGBA : GL_RGB);
160 glTexImage2D( GL_TEXTURE_2D, 0, STORAGE_TYPE, imageWidth, imageHeight, 0, STORAGE_TYPE, GL_UNSIGNED_BYTE, data);
161 glGenerateMipmap(GL_TEXTURE_2D);
162 printf( "[INFO]: Successfully loaded texture \"%s\" with handle %d\n", filename, texHandle );
163
164 return texHandle;
165}
